#include <stdio.h>

#include <stdlib.h>

#define length  100



int main()

{

  int i,j=0;

  for(i=0;i<length;i++){

    for(j=0;j<i;j++)

    {

        printf(" ");

    }

    printf("*");

    system("CLS");

  }

  for(i=length;i>0;i--)

  {

      for(j=i;j>0;j--)

      {

          printf(" ");

      }

      printf("*");

      system("CLS");

  }
